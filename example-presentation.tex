% Bayes' Law Slide
\begin{frame}{Bayes' Law}
\[
P(A \mid B) = \frac{P(B \mid A) \cdot P(A)}{P(B)}
\]
\pause
\begin{block}{Interpretation}
Bayesâ€™ Law allows us to update our beliefs about \(A\) after observing \(B\), assuming we know:
\begin{itemize}
  \item \(P(A)\): prior belief
  \item \(P(B \mid A)\): likelihood
  \item \(P(B)\): evidence
\end{itemize}
\end{block}
\end{frame}
